import gensim.downloader as api
m=api.load("glove-wiki-gigaword-100")
prompt="Describe the beautiful landscape during the sunset."
def enrich_prompt(prompt,m,n=5):
    words=prompt.split()
    enriched_res=[]
    for word in words:
        lower_words=word.lower()
        if lower_words in  m:
            similar_words=m.most_similar(lower_words,topn=n)
            similar_words_list=[w[0] for w in similar_words]
            enriched_res.append(" ".join(similar_words_list))
        else:
            enriched_res.append(" ".join(word))
    return " ".join(enriched_res)
enriched_prompt = enrich_prompt(prompt,m)

print("Original Prompt:")
print(prompt)

print("\nEnriched Prompt:")
print(enriched_prompt)
